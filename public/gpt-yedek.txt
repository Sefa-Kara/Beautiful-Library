<body>
    <div class="gallery-container" id="gallery">
        <div class="scroll-indicator">
            Scroll Down → Move Right | Scroll Up → Move Left
        </div>
        
        <div class="gallery-track" id="track">
            <!-- 29 pages will be generated by JavaScript -->
        </div>
        
        <div class="position-indicator" id="position">
            Page 1 of 29
        </div>
        
        <div class="progress-bar" id="progress"></div>
    </div>

    <script>
        class BookshelfGridGallery {
            constructor() {
                this.container = document.getElementById('gallery');
                this.track = document.getElementById('track');
                this.positionIndicator = document.getElementById('position');
                this.progressBar = document.getElementById('progress');
                
                this.currentPosition = 0;
                this.totalPages = 29;
                this.scrollSensitivity = 25; // Reduced for smoother scrolling
                this.isScrolling = false;
                this.maxScroll = (this.totalPages - 1) * 100;
                
                this.init();
            }
            
            init() {
                this.createPages();
                this.addEventListeners();
                this.updateGallery();
            }
            
            createPages() {
                for (let pageIndex = 0; pageIndex < this.totalPages; pageIndex++) {
                    const page = document.createElement('div');
                    page.className = 'gallery-page';
                    
                    // Create the main grid for this page
                    const grid = this.createBookshelfGrid(pageIndex);
                    page.appendChild(grid);
                    
                    // Create side panels showing adjacent pages
                    /*if (pageIndex > 0) {
                        const leftPanel = document.createElement('div');
                        leftPanel.className = 'side-panel left';
                        const leftGrid = this.createBookshelfGrid(pageIndex - 1);
                        leftPanel.appendChild(leftGrid);
                        page.appendChild(leftPanel);
                    }*/
                    
                    // if (pageIndex < this.totalPages - 1) {
                    //     const rightPanel = document.createElement('div');
                    //     rightPanel.className = 'side-panel right';
                    //     const rightGrid = this.createBookshelfGrid(pageIndex + 1);
                    //     rightPanel.appendChild(rightGrid);
                    //     page.appendChild(rightPanel);
                    // }
                    
                    this.track.appendChild(page);
                }
                this.track.style.width = `${this.totalPages * 100}vw`; // because each page + margin = 100vw total
            }
            
            createBookshelfGrid(pageIndex) {
                const grid = document.createElement('div');
                grid.className = 'bookshelf-grid';
                
                // Add page number overlay
                const pageNumber = document.createElement('div');
                pageNumber.className = 'page-number';
                pageNumber.textContent = `Page ${pageIndex + 1}`;
                grid.appendChild(pageNumber);
                
                // Create 10 bookshelf items (5 rows × 2 columns)
                for (let i = 0; i < 10; i++) {
                    const shelf = document.createElement('div');
                    shelf.className = 'bookshelf-item bookshelf-bg';
                    
                    // Add wood grain overlay
                    const woodGrain = document.createElement('div');
                    woodGrain.className = 'wood-grain';
                    shelf.appendChild(woodGrain);
                    
                    grid.appendChild(shelf);
                }
                
                return grid;
            }
            
            addEventListeners() {
                // Mouse wheel event
                this.container.addEventListener('wheel', (e) => {
                    e.preventDefault();
                    this.handleWheel(e);
                });
                
                // Keyboard navigation
                document.addEventListener('keydown', (e) => {
                    if (e.key === 'ArrowLeft') this.moveLeft();
                    if (e.key === 'ArrowRight') this.moveRight();
                    if (e.key === 'Home') this.goToPosition(0);
                    if (e.key === 'End') this.goToPosition(this.maxScroll);
                });
                
                // Touch events for mobile
                let startX = 0;
                let endX = 0;
                
                this.container.addEventListener('touchstart', (e) => {
                    startX = e.touches[0].clientX;
                });
                
                this.container.addEventListener('touchmove', (e) => {
                    e.preventDefault();
                });
                
                this.container.addEventListener('touchend', (e) => {
                    endX = e.changedTouches[0].clientX;
                    this.handleSwipe(startX, endX);
                });
            }
            
            handleWheel(e) {
                if (this.isScrolling) return;
                
                const delta = e.deltaY;
                
                // Scroll down = move right, Scroll up = move left
                if (delta > 0) {
                    this.moveRight();
                } else if (delta < 0) {
                    this.moveLeft();
                }
            }
            
            handleSwipe(startX, endX) {
                const threshold = 50;
                const difference = startX - endX;
                
                if (Math.abs(difference) > threshold) {
                    if (difference > 0) {
                        this.moveRight();
                    } else {
                        this.moveLeft();
                    }
                }
            }
            
            moveLeft() {
                this.currentPosition = Math.max(0, this.currentPosition - this.scrollSensitivity);
                this.updateGallery();
            }
            
            moveRight() {
                this.currentPosition = Math.min(this.maxScroll, this.currentPosition + this.scrollSensitivity);
                this.updateGallery();
            }
            
            goToPosition(position) {
                this.currentPosition = Math.max(0, Math.min(this.maxScroll, position));
                this.updateGallery();
            }
            
            updateGallery() {
                this.isScrolling = true;
                
                // Move the track
                const translateX = -this.currentPosition;
                this.track.style.transform = `translateX(${translateX}vw)`;
                
                // Update page states for scaling effects
                this.updatePageStates();
                
                // Update position indicator
                const currentPage = Math.floor(this.currentPosition / 100) + 1;
                this.positionIndicator.textContent = `Page ${currentPage} of ${this.totalPages}`;
                
                // Update progress bar
                const progress = (this.currentPosition / this.maxScroll) * 100;
                this.progressBar.style.width = `${progress}%`;
                
                // Reset scrolling flag after animation
                setTimeout(() => {
                    this.isScrolling = false;
                }, 600); // Increased to match animation duration
            }
            
            updatePageStates() {
                const pages = document.querySelectorAll('.gallery-page');
                const currentPageIndex = Math.round(this.currentPosition / 100);
                
                pages.forEach((page, index) => {
                    page.classList.remove('active', 'adjacent');
                    
                    if (index === currentPageIndex) {
                        page.classList.add('active');
                    } else if (Math.abs(index - currentPageIndex) === 1) {
                        page.classList.add('adjacent');
                    }
                });
            }
        }
        
        // Initialize the gallery
        document.addEventListener('DOMContentLoaded', () => {
            new BookshelfGridGallery();
        });
    </script>
